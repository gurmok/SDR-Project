cmake_minimum_required(VERSION 2.8.3)
project(sdr_tf_publisher)

find_package(catkin REQUIRED COMPONENTS
  geometry_msgs
  roscpp
  tf
)

catkin_package(
  INCLUDE_DIRS include
  LIBRARIES pagv_tf_publisher
  CATKIN_DEPENDS geometry_msgs roscpp tf
  DEPENDS system_lib
)

include_directories(
  ${catkin_INCLUDE_DIRS}
)

add_executable(sdr_tf_publisher_node src/sdr_tf_publisher_node.cpp)

 target_link_libraries(sdr_tf_publisher_node
   ${catkin_LIBRARIES}
 )
